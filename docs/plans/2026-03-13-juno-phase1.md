# Juno Phase 1 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build MVP of AI-driven English learning app with flashcard learning, SRS database, and LLM story generation.

**Architecture:** MVVM + Clean Architecture with Jetpack Compose. UI Layer → Domain Layer → Data Layer. Repository pattern for data access.

**Tech Stack:**
- UI: Jetpack Compose + Material 3
- Navigation: Compose Navigation
- DI: Hilt
- Local DB: Room + DataStore
- Backend: Firebase (Auth + Firestore + Functions) or Supabase
- AI: OpenAI/DeepSeek API via Firebase Functions

---

## Project Overview

### Core Features (Phase 1)
1. **Flashcard Learning** - Card-based word memorization (like 百词斩)
2. **SRS Database** - Spaced Repetition System based on SM-2 algorithm
3. **LLM Story Generation** - AI-generated contextual stories from word list

### Target Screens
1. Home/Flashcard Screen - Main learning interface
2. Word List Management - Add/Edit/Delete words
3. Review Session - SRS-based review flow
4. Story Reading - LLM-generated contextual stories
5. Settings/Profile - User preferences and progress

package com.juno.app.ui.theme

import android.app.Activity
import android.os.Build
import androidx.compose.foundation.isSystemInDarkTheme
import androidx.compose.material3.MaterialTheme
import androidx.compose.material3.darkColorScheme
import androidx.compose.material3.dynamicDarkColorScheme
import androidx.compose.material3.dynamicLightColorScheme
import androidx.compose.material3.lightColorScheme
import androidx.compose.runtime.Composable
import androidx.compose.runtime.SideEffect
import androidx.compose.ui.graphics.toArgb
import androidx.compose.ui.platform.LocalContext
import androidx.compose.ui.platform.LocalView
import androidx.core.view.WindowCompat

private val LightColorScheme = lightColorScheme(
    primary = Primary,
    onPrimary = OnPrimary,
    primaryContainer = PrimaryLight,
    onPrimaryContainer = PrimaryDark,
    secondary = Secondary,
    onSecondary = OnSecondary,
    secondaryContainer = SecondaryLight,
    onSecondaryContainer = SecondaryDark,
    tertiary = Tertiary,
    onTertiary = OnPrimary,
    tertiaryContainer = TertiaryLight,
    onTertiaryContainer = TertiaryDark,
    background = Background,
    onBackground = OnBackground,
    surface = Surface,
    onSurface = OnSurface,
    surfaceVariant = CardBackground,
    onSurfaceVariant = OnSurface,
    error = Error,
    onError = OnPrimary
)

private val DarkColorScheme = darkColorScheme(
    primary = PrimaryLight,
    onPrimary = PrimaryDark,
    primaryContainer = PrimaryDark,
    onPrimaryContainer = PrimaryLight,
    secondary = SecondaryLight,
    onSecondary = SecondaryDark,
    secondaryContainer = SecondaryDark,
    onSecondaryContainer = SecondaryLight,
    tertiary = TertiaryLight,
    onTertiary = TertiaryDark,
    tertiaryContainer = TertiaryDark,
    onTertiaryContainer = TertiaryLight,
    background = BackgroundDark,
    onBackground = OnBackgroundDark,
    surface = SurfaceDark,
    onSurface = OnSurfaceDark,
    surfaceVariant = CardBackgroundDark,
    onSurfaceVariant = OnSurfaceDark,
    error = Error,
    onError = OnPrimary
)

@Composable
fun JunoTheme(
    darkTheme: Boolean = isSystemInDarkTheme(),
    dynamicColor: Boolean = false,
    content: @Composable () -> Unit
) {
    val colorScheme = when {
        dynamicColor && Build.VERSION.SDK_INT >= Build.VERSION_CODES.S -> {
            val context = LocalContext.current
            if (darkTheme) dynamicDarkColorScheme(context) else dynamicLightColorScheme(context)
        }
        darkTheme -> DarkColorScheme
        else -> LightColorScheme
    }

    val view = LocalView.current
    if (!view.isInEditMode) {
        SideEffect {
            val window = (view.context as Activity).window
            window.statusBarColor = colorScheme.background.toArgb()
            WindowCompat.getInsetsController(window, view).isAppearanceLightStatusBars = !darkTheme
        }
    }

    MaterialTheme(
        colorScheme = colorScheme,
        typography = Typography,
        content = content
    )
}

package com.juno.app.data.local.dao

import androidx.room.*
import com.juno.app.data.local.entity.WordEntity
import kotlinx.coroutines.flow.Flow

@Dao
interface WordDao {
    @Query("SELECT * FROM words ORDER BY createdAt DESC")
    fun getAllWords(): Flow<List<WordEntity>>

    @Query("SELECT * FROM words WHERE isLearned = 0 ORDER BY createdAt DESC")

    @Query("SELECT * FROM words WHERE word LIKE '%' || :query || '%' OR meaning LIKE '%' || :query || '%'")

// ReviewRepositoryImpl.kt - SM-2 Algorithm
package com.juno.app.data.repository

import com.juno.app.data.local.dao.ReviewDao
import com.juno.app.data.local.dao.WordDao
import com.juno.app.data.local.entity.ReviewRecordEntity
import com.juno.app.domain.repository.ReviewRepository
import kotlinx.coroutines.flow.Flow
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class ReviewRepositoryImpl @Inject constructor(
    private val reviewDao: ReviewDao,
    private val wordDao: WordDao
) : ReviewRepository {

    override suspend fun getReviewRecord(wordId: Long): ReviewRecordEntity? {
        return reviewDao.getReviewRecord(wordId)
    }

    override suspend fun getDueReviews(): List<ReviewRecordEntity> {
        return reviewDao.getDueReviews()
    }

    override fun getDueReviewsFlow(): Flow<List<ReviewRecordEntity>> {
        return reviewDao.getDueReviewsFlow()
    }

    override fun getDueReviewCount(): Flow<Int> {
        return reviewDao.getDueReviewCount()
    }

    override suspend fun createReviewRecord(wordId: Long): Long {
        val record = ReviewRecordEntity(wordId = wordId)
        return reviewDao.insertReviewRecord(record)
    }

    override suspend fun updateReviewRecord(record: ReviewRecordEntity) {
        reviewDao.updateReviewRecord(record)
    }

    /**
     * SM-2 Algorithm Implementation
     * 
     * Key formulas:
     * EF' = EF + (0.1 - (5-q) * (0.08 + (5-q) * 0.02))
     * where EF is the easiness factor, q is quality (0-5)
     * 
     * Interval calculation:
     * If q < 3: interval = 1 (start over)
     * If repetitions = 1: interval = 1
     * If repetitions = 2: interval = 6
     * Otherwise: interval = previous_interval * EF
     */
    override suspend fun processReview(wordId: Long, quality: Int) {
        val existingRecord = reviewDao.getReviewRecord(wordId)
        
        if (existingRecord == null) {
            // Create new record
            val newRecord = ReviewRecordEntity(
                wordId = wordId,
                easeFactor = 2.5f,
                interval = 1,
                repetitions = 1,
                nextReviewDate = System.currentTimeMillis() + dayToMillis(1),
                lastReviewDate = System.currentTimeMillis(),
                reviewCount = 1,
                correctCount = if (quality >= 3) 1 else 0
            )
            reviewDao.insertReviewRecord(newRecord)
            return
        }

        // Calculate new ease factor
        val newEaseFactor = calculateNewEaseFactor(existingRecord.easeFactor, quality)
        
        // Calculate new interval and repetitions
        val (newInterval, newRepetitions) = if (quality < 3) {
            // Failed - reset
            Pair(1, 0)
        } else {
            val newReps = existingRecord.repetitions + 1
            val interval = when (newReps) {
                1 -> 1
                2 -> 6
                else -> (existingRecord.interval * newEaseFactor).toInt()
            }
            Pair(interval, newReps)
        }

        val updatedRecord = existingRecord.copy(
            easeFactor = newEaseFactor,
            interval = newInterval,
            repetitions = newRepetitions,
            nextReviewDate = System.currentTimeMillis() + dayToMillis(newInterval),
            lastReviewDate = System.currentTimeMillis(),
            reviewCount = existingRecord.reviewCount + 1,
            correctCount = if (quality >= 3) existingRecord.correctCount + 1 else existingRecord.correctCount
        )

        reviewDao.updateReviewRecord(updatedRecord)
    }

    private fun calculateNewEaseFactor(currentEF: Float, quality: Int): Float {
        val newEF = currentEF + (0.1f - (5 - quality) * (0.08f + (5 - quality) * 0.02f))
        return maxOf(1.3f, newEF) // Minimum EF is 1.3
    }

    private fun dayToMillis(days: Int): Long = days * 24 * 60 * 60 * 1000L
}

data class FlashcardUiState(
    val isLoading: Boolean = true,
    val currentIndex: Int = 0,
    val words: List<WordEntity> = emptyList(),
    val isFlipped: Boolean = false,
    val isComplete: Boolean = false,
    val dailyGoalCompleted: Int = 0
)

@HiltViewModel
class FlashcardViewModel @Inject constructor(
    private val wordRepository: WordRepository,
    private val reviewRepository: ReviewRepository
) : ViewModel() {

    private val _uiState = MutableStateFlow(FlashcardUiState())
    val uiState: StateFlow<FlashcardUiState> = _uiState.asStateFlow()

    init {
        loadWords()
    }

    private fun loadWords() {
        viewModelScope.launch {
            wordRepository.getUnlearnedWords().collect { words ->
                _uiState.update {
                    it.copy(
                        isLoading = false,
                        words = words.shuffled().take(20), // Daily batch
                        isComplete = words.isEmpty()
                    )
                }
            }
        }
    }

    fun flipCard() {
        _uiState.update { it.copy(isFlipped = !it.isFlipped) }
    }

    fun nextWord(remembered: Boolean) {
        viewModelScope.launch {
            val currentWord = _uiState.value.words.getOrNull(_uiState.value.currentIndex)
            
            currentWord?.let { word ->
                // Create review record if not exists
                if (reviewRepository.getReviewRecord(word.id) == null) {
                    reviewRepository.createReviewRecord(word.id)
                }
                
                // Process review with SM-2 quality
                val quality = if (remembered) 4 else 1
                reviewRepository.processReview(word.id, quality)
                
                // Update learned status if remembered
                if (remembered) {
                    wordRepository.updateWord(word.copy(isLearned = true))
                }
            }

            _uiState.update { state ->
                val nextIndex = state.currentIndex + 1
                if (nextIndex >= state.words.size) {
                    state.copy(isComplete = true, isFlipped = false)
                } else {
                    state.copy(
                        currentIndex = nextIndex,
                        isFlipped = false,
                        dailyGoalCompleted = state.dailyGoalCompleted + 1
                    )
                }
            }
        }
    }

    fun restart() {
        loadWords()
    }
}

import kotlinx.coroutines.launch
import javax.inject.Inject

data class ReviewUiState(
    val isLoading: Boolean = true,
    val currentReviewIndex: Int = 0,
    val dueReviews: List<WordEntity> = emptyList(),
    val isFlipped: Boolean = false,
    val isComplete: Boolean = false,
    val reviewedCount: Int = 0
)

@HiltViewModel
class ReviewViewModel @Inject constructor(
    private val reviewRepository: ReviewRepository,
    private val wordRepository: WordRepository
) : ViewModel() {

    private val _uiState = MutableStateFlow(ReviewUiState())
    val uiState: StateFlow<ReviewUiState> = _uiState.asStateFlow()

    init {
        loadDueReviews()
    }

    private fun loadDueReviews() {
        viewModelScope.launch {
            val reviews = reviewRepository.getDueReviews()
            val words = reviews.mapNotNull { review ->
                wordRepository.getWordById(review.wordId)
            }
            _uiState.update {
                it.copy(
                    isLoading = false,
                    dueReviews = words,
                    isComplete = words.isEmpty()
                )
            }
        }
    }

    fun flipCard() {
        _uiState.update { it.copy(isFlipped = !it.isFlipped) }
    }

    fun processReview(quality: Int) {
        viewModelScope.launch {
            val currentWord = _uiState.value.dueReviews.getOrNull(_uiState.value.currentReviewIndex)
            
            currentWord?.let { word ->
                reviewRepository.processReview(word.id, quality)
            }

            _uiState.update { state ->
                val nextIndex = state.currentReviewIndex + 1
                if (nextIndex >= state.dueReviews.size) {
                    state.copy(isComplete = true, isFlipped = false)
                } else {
                    state.copy(
                        currentReviewIndex = nextIndex,
                        isFlipped = false,
                        reviewedCount = state.reviewedCount + 1
                    )
                }
            }
        }
    }

    // Quality ratings: 0-5
    // 0-2: Failed (will review sooner)
    // 3-5: Passed (increase interval)
    fun markAgain() = processReview(1)
    fun markHard() = processReview(3)
    fun markGood() = processReview(4)
    fun markEasy() = processReview(5)

    fun restart() {
        loadDueReviews()
    }
}
