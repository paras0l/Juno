```kotlin
package com.juno.app.data.repository

import android.content.Context
import android.speech.tts.TextToSpeech
import dagger.hilt.android.qualifiers.ApplicationContext
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import java.util.Locale
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class TtsService @Inject constructor(
    @ApplicationContext private val context: Context
) {
    private var tts: TextToSpeech? = null
    
    private val _isReady = MutableStateFlow(false)
    val isReady: StateFlow<Boolean> = _isReady.asStateFlow()
    
    private val _isPlaying = MutableStateFlow(false)
    val isPlaying: StateFlow<Boolean> = _isPlaying.asStateFlow()
    
    private var speechRate = 1.0f
    
    init {
        tts = TextToSpeech(context) { status ->
            if (status == TextToSpeech.SUCCESS) {
                tts?.language = Locale.US
                _isReady.value = true
            }
        }
    }
    
    fun speak(text: String) {
        if (_isReady.value) {
            _isPlaying.value = true
            tts?.setSpeechRate(speechRate)
            tts?.speak(text, TextToSpeech.QUEUE_FLUSH, null, "juno_tts")
        }
    }
    
    fun stop() {
        tts?.stop()
        _isPlaying.value = false
    }
    
    fun setSpeechRate(rate: Float) {
        speechRate = rate.coerceIn(0.5f, 2.0f)
    }
    
    fun destroy() {
        tts?.stop()
        tts?.shutdown()
        tts = null
    }
}
```

```kotlin
    @ApplicationContext private val context: Context
```

```kotlin
    private var recorder: MediaRecorder? = null
    private var outputFile: File? = null
    
    private val _isRecording = MutableStateFlow(false)
    val isRecording: StateFlow<Boolean> = _isRecording.asStateFlow()
    
    private val _amplitude = MutableStateFlow(0)
    val amplitude: StateFlow<Int> = _amplitude.asStateFlow()
    
    fun startRecording(): String {
        val file = File(context.cacheDir, "recording_${System.currentTimeMillis()}.m4a")
        outputFile = file
        
        recorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            MediaRecorder(context)
        } else {
            @Suppress("DEPRECATION")
            MediaRecorder()
        }.apply {
            setAudioSource(MediaRecorder.AudioSource.MIC)
            setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
            setAudioEncoder(MediaRecorder.AudioEncoder.AAC)
            setAudioEncodingBitRate(128000)
            setAudioSamplingRate(44100)
            setOutputFile(file.absolutePath)
            prepare()
            start()
        }
        
        _isRecording.value = true
        return file.absolutePath
    }
    
    fun stopRecording(): File? {
        recorder?.apply {
            stop()
            release()
        }
        recorder = null
        _isRecording.value = false
        return outputFile
    }
    
    fun cancelRecording() {
        recorder?.apply {
            stop()
            release()
        }
        recorder = null
        _isRecording.value = false
        outputFile?.delete()
        outputFile = null
    }
    
    fun getMaxAmplitude(): Int {
        return recorder?.maxAmplitude ?: 0
    }
}
```

```kotlin
package com.juno.app.data.remote

import android.content.Context
import dagger.hilt.android.qualifiers.ApplicationContext
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.MultipartBody
import okhttp3.OkHttpClient
import okhttp3.Request
import okhttp3.RequestBody.Companion.asRequestBody
import java.io.File
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class SpeechRecognitionService @Inject constructor(
    @ApplicationContext private val context: Context,
    private val okHttpClient: OkHttpClient
) {
    data class RecognitionResult(
        val success: Boolean,
        val text: String = "",
        val confidence: Float = 0f,
        val errorMessage: String? = null
    )
    
    /**
     * 语音识别 - 使用OpenAI Whisper API
     * 注意：实际部署需要通过后端调用，避免暴露API Key
     */
    suspend fun recognizeAudio(audioFile: File, apiKey: String): RecognitionResult = 
        withContext(Dispatchers.IO) {
            try {
                val requestBody = MultipartBody.Builder()
                    .setType(MultipartBody.FORM)
                    .addFormDataPart(
                        "file",
                        audioFile.name,
                        audioFile.asRequestBody("audio/m4a".toMediaType())
                    )
                    .addFormDataPart("model", "whisper-1")
                    .addFormDataPart("language", "en")
                    .build()
                
                val request = Request.Builder()
                    .url("https://api.openai.com/v1/audio/transcriptions")
                    .addHeader("Authorization", "Bearer $apiKey")
                    .post(requestBody)
                    .build()
                
                val response = okHttpClient.newCall(request).execute()
                
                if (response.isSuccessful) {
                    val responseBody = response.body?.string()
                    // Parse JSON response
                    // {"text": "transcribed text"}
                    val text = parseWhisperResponse(responseBody ?: "")
                    RecognitionResult(success = true, text = text)
                } else {
                    RecognitionResult(
                        success = false,
                        errorMessage = "API Error: ${response.code}"
                    )
                }
            } catch (e: Exception) {
                RecognitionResult(
                    success = false,
                    errorMessage = e.message
                )
            }
        }
    
    private fun parseWhisperResponse(json: String): String {
        // Simple JSON parsing for {"text": "..."}
        val regex = """"text"\s*:\s*"([^"]*)"""".toRegex()
        return regex.find(json)?.groupValues?.get(1) ?: ""
    }
    
    /**
     * 本地语音识别（使用Android内置）
     * 不需要网络，但准确度较低
     */
    fun startLocalRecognition() {
        // 使用SpeechRecognizer进行本地识别
        // 略 - 需要更复杂的实现
    }
}
```

```kotlin
package com.juno.app.data.remote

import android.content.Context
import dagger.hilt.android.qualifiers.ApplicationContext
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import okhttp3.OkHttpClient
import okhttp3.Request
import org.jsoup.Jsoup
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class LinkParserService @Inject constructor(
    @ApplicationContext private val context: Context,
    private val okHttpClient: OkHttpClient
) {
    data class ParsedContent(
        val title: String,
        val text: String,
        val sourceUrl: String,
        val extractedWords: List<String> = emptyList()
    )
    
    /**
     * 解析URL并提取文本内容
     */
    suspend fun parseUrl(url: String): Result<ParsedContent> = withContext(Dispatchers.IO) {
        try {
            val request = Request.Builder()
                .url(url)
                .header("User-Agent", "Mozilla/5.0")
                .build()
            
            val response = okHttpClient.newCall(request).execute()
            
            if (!response.isSuccessful) {
                return@withContext Result.failure(Exception("Failed to fetch: ${response.code}"))
            }
            
            val html = response.body?.string() ?: return@withContext Result.failure(Exception("Empty response"))
            
            // 使用Jsoup解析HTML
            val document = Jsoup.parse(html)
            
            // 提取纯文本
            val text = document.body().text()
            
            // 提取标题
            val title = document.title().ifEmpty { 
                document.select("meta[property=og:title]").attr("content")
            }
            
            Result.success(ParsedContent(
                title = title,
                text = text.take(10000), // 限制长度
                sourceUrl = url
            ))
        } catch (e: Exception) {
            Result.failure(e)
        }
    }
    
    /**
```

```kotlin
package com.juno.app.data.remote

import android.content.Intent
import dagger.hilt.android.qualifiers.ApplicationContext
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class ShareReceiverManager @Inject constructor(
    @ApplicationContext private val context: android.content.Context,
    private val linkParserService: LinkParserService
) {
    data class SharedContent(
        val type: ContentType,
        val url: String? = null,
        val text: String? = null
    )
    
    enum class ContentType {
        URL,
        TEXT,
        UNKNOWN
    }
    
    fun parseIntent(intent: Intent?): SharedContent? {
        return when (intent?.action) {
            Intent.ACTION_SEND -> {
                if (intent.type == "text/plain") {
                    val text = intent.getStringExtra(Intent.EXTRA_TEXT)
                    when {
                        text.isNullOrBlank() -> null
                        text.startsWith("http://") || text.startsWith("https://") -> {
                            SharedContent(type = ContentType.URL, url = text)
                        }
                        else -> {
                            SharedContent(type = ContentType.TEXT, text = text)
                        }
                    }
                } else null
            }
            else -> null
        }
    }
}
```

```kotlin
import android.accessibilityservice.AccessibilityService
import android.view.accessibility.AccessibilityEvent
import dagger.hilt.android.AndroidEntryPoint

@AndroidEntryPoint
class OcrAccessibilityService : AccessibilityService() {
    
    private var listener: OcrListener? = null
    
    interface OcrListener {
        fun onTextDetected(text: String)
    }
    
    fun setListener(listener: OcrListener?) {
        this.listener = listener
    }
    
    override fun onAccessibilityEvent(event: AccessibilityEvent?) {
        event?.let {
            if (it.eventType == AccessibilityEvent.TYPE_VIEW_TEXT_CHANGED ||
                it.eventType == AccessibilityEvent.TYPE_VIEW_FOCUSED) {
                val text = it.text?.joinToString(" ") ?: return
                if (text.isNotBlank() && text.length > 2) {
                    listener?.onTextDetected(text)
                }
            }
        }
    }
    
    override fun onInterrupt() {
        // Service interrupted
    }
    
    override fun onServiceConnected() {
        super.onServiceConnected()
        // Service connected
    }
}
```

```kotlin
package com.juno.app.service

import android.app.Notification
import android.app.NotificationChannel
import android.app.NotificationManager
import android.app.PendingIntent
import android.app.Service
import android.content.Context
import android.content.Intent
import android.graphics.PixelFormat
import android.os.Build
import android.os.IBinder
import android.view.Gravity
import android.view.LayoutInflater
import android.view.WindowManager
import android.widget.Button
import android.widget.TextView
import androidx.core.app.NotificationCompat
import com.juno.app.MainActivity
import com.juno.app.R
import dagger.hilt.android.AndroidEntryPoint

@AndroidEntryPoint
class FloatingWindowService : Service() {
    
    private var windowManager: WindowManager? = null
    private var floatingView: android.view.View? = null
    
    companion object {
        const val CHANNEL_ID = "floating_window_channel"
        const val NOTIFICATION_ID = 1001
        
        const val ACTION_START = "com.juno.app.action.START_FLOATING"
        const val ACTION_STOP = "com.juno.app.action.STOP_FLOATING"
        
        fun start(context: Context) {
            val intent = Intent(context, FloatingWindowService::class.java).apply {
                action = ACTION_START
            }
            context.startForegroundService(intent)
        }
        
        fun stop(context: Context) {
            val intent = Intent(context, FloatingWindowService::class.java).apply {
                action = ACTION_STOP
            }
            context.startService(intent)
        }
    }
    
    override fun onBind(intent: Intent?): IBinder? = null
    
    override fun onCreate() {
        super.onCreate()
        windowManager = getSystemService(WINDOW_SERVICE) as WindowManager
        createNotificationChannel()
    }
    
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        when (intent?.action) {
            ACTION_START -> {
                startForeground(NOTIFICATION_ID, createNotification())
                showFloatingWindow()
            }
            ACTION_STOP -> {
                hideFloatingWindow()
                stopForeground(STOP_FOREGROUND_REMOVE)
                stopSelf()
            }
        }
        return START_STICKY
    }
    
    private fun createNotificationChannel() {
        val channel = NotificationChannel(
            CHANNEL_ID,
            "悬浮窗服务",
            NotificationManager.IMPORTANCE_LOW
        ).apply {
            description = "用于OCR取词"
        }
        val notificationManager = getSystemService(NotificationManager::class.java)
        notificationManager.createNotificationChannel(channel)
    }
    
    private fun createNotification(): Notification {
        val pendingIntent = PendingIntent.getActivity(
            this,
            0,
            Intent(this, MainActivity::class.java),
            PendingIntent.FLAG_IMMUTABLE
        )
        
        return NotificationCompat.Builder(this, CHANNEL_ID)
            .setContentTitle("Juno OCR")
            .setContentText("点击返回应用")
            .setSmallIcon(android.R.drawable.ic_menu_compass)
            .setContentIntent(pendingIntent)
            .build()
    }
    
    private fun showFloatingWindow() {
        if (floatingView != null) return
        
        val params = WindowManager.LayoutParams(
            WindowManager.LayoutParams.WRAP_CONTENT,
            WindowManager.LayoutParams.WRAP_CONTENT,
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                WindowManager.LayoutParams.TYPE_APPLICATION_OVERLAY
            } else {
                @Suppress("DEPRECATION")
                WindowManager.LayoutParams.TYPE_PHONE
            },
            WindowManager.LayoutParams.FLAG_NOT_FOCUSABLE,
            PixelFormat.TRANSLUCENT
        ).apply {
            gravity = Gravity.TOP or Gravity.END
            x = 0
            y = 200
        }
        
        floatingView = LayoutInflater.from(this).inflate(R.layout.floating_window, null)
        
        floatingView?.findViewById<Button>(R.id.btnCapture)?.setOnClickListener {
            // 触发OCR识别
            // 可以调用ML Kit进行文字识别
        }
        
        floatingView?.findViewById<Button>(R.id.btnClose)?.setOnClickListener {
            stop(this)
        }
        
        windowManager?.addView(floatingView, params)
    }
    
    private fun hideFloatingWindow() {
        floatingView?.let {
            windowManager?.removeView(it)
            floatingView = null
        }
    }
    
    override fun onDestroy() {
        hideFloatingWindow()
        super.onDestroy()
    }
}
```

```kotlin
package com.juno.app.ui.screens.pronunciation

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.juno.app.data.local.AudioPlaybackManager
import com.juno.app.data.local.VoiceRecordingManager
import com.juno.app.data.repository.TtsService
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.launch
import java.io.File
import javax.inject.Inject

data class PronunciationUiState(
    val word: String = "",
    val phonetic: String = "",
    val example: String = "",
    val isPlayingOriginal: Boolean = false,
    val isRecording: Boolean = false,
    val recognizedText: String = "",
    val score: Int = 0,
    val feedback: String = "",
    val error: String? = null
)

@HiltViewModel
class PronunciationViewModel @Inject constructor(
    private val ttsService: TtsService,
    private val audioPlaybackManager: AudioPlaybackManager,
    private val voiceRecordingManager: VoiceRecordingManager
) : ViewModel() {
    
    private val _uiState = MutableStateFlow(PronunciationUiState())
    val uiState: StateFlow<PronunciationUiState> = _uiState.asStateFlow()
    
    private var recordingFile: File? = null
    
    fun setWord(word: String, example: String, phonetic: String = "") {
        _uiState.value = _uiState.value.copy(
            word = word,
            phonetic = phonetic,
            example = example
        )
    }
    
    fun playOriginal() {
        _uiState.value = _uiState.value.copy(isPlayingOriginal = true)
        ttsService.speak(_uiState.value.word)
        
        viewModelScope.launch {
            kotlinx.coroutines.delay(2000)
            _uiState.value = _uiState.value.copy(isPlayingOriginal = false)
        }
    }
    
    fun startRecording() {
        recordingFile = File(voiceRecordingManager.startRecording())
        _uiState.value = _uiState.value.copy(isRecording = true)
    }
    
    fun stopRecording() {
        voiceRecordingManager.stopRecording()
        _uiState.value = _uiState.value.copy(isRecording = false)
        
        // 这里应该调用语音识别API
        // 暂时使用模拟结果
        _uiState.value = _uiState.value.copy(
            recognizedText = _uiState.value.word, // 模拟识别
            score = calculateScore(_uiState.value.word, _uiState.value.word),
            feedback = getFeedback(calculateScore(_uiState.value.word, _uiState.value.word))
        )
    }
    
    private fun calculateScore(original: String, recognized: String): Int {
        if (original.equals(recognized, ignoreCase = true)) return 100
        // 简单的相似度计算
        val distance = levenshteinDistance(original.lowercase(), recognized.lowercase())
        val maxLen = maxOf(original.length, recognized.length)
        return ((maxLen - distance) * 100 / maxLen).coerceIn(0, 100)
    }
    
    private fun levenshteinDistance(s1: String, s2: String): Int {
        val dp = Array(s1.length + 1) { IntArray(s2.length + 1) }
        for (i in 0..s1.length) dp[i][0] = i
        for (j in 0..s2.length) dp[0][j] = j
        
        for (i in 1..s1.length) {
            for (j in 1..s2.length) {
                val cost = if (s1[i-1] == s2[j-1]) 0 else 1
                dp[i][j] = minOf(
                    dp[i-1][j] + 1,
                    dp[i][j-1] + 1,
                    dp[i-1][j-1] + cost
                )
            }
        }
        return dp[s1.length][s2.length]
    }
    
    private fun getFeedback(score: Int): String = when {
        score >= 90 -> "发音非常棒！"
        score >= 70 -> "发音不错，继续加油！"
        score >= 50 -> "有一些发音需要改进"
        else -> "建议多听几遍原音"
    }
}
```
